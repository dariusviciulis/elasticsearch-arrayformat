001package org.xbib.elasticsearch.rest.arrayformat;
002
003import org.elasticsearch.action.search.SearchRequest;
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.rest.BaseRestHandler;
008import org.elasticsearch.rest.RestChannel;
009import org.elasticsearch.rest.RestController;
010import org.elasticsearch.rest.RestRequest;
011import org.elasticsearch.rest.action.search.RestSearchAction;
012
013import static org.elasticsearch.rest.RestRequest.Method.GET;
014import static org.elasticsearch.rest.RestRequest.Method.POST;
015
016public class ArrayformatRestSearchAction extends BaseRestHandler {
017
018    @Inject
019    public ArrayformatRestSearchAction(Settings settings, Client client, RestController controller) {
020        super(settings, client);
021        controller.registerHandler(GET, "/_search_arrayformat", this);
022        controller.registerHandler(POST, "/_search_arrayformat", this);
023        controller.registerHandler(GET, "/{index}/_search_arrayformat", this);
024        controller.registerHandler(POST, "/{index}/_search_arrayformat", this);
025        controller.registerHandler(GET, "/{index}/{type}/_search_arrayformat", this);
026        controller.registerHandler(POST, "/{index}/{type}/_search_arrayformat", this);
027        controller.registerHandler(GET, "/_search_arrayformat/template", this);
028        controller.registerHandler(POST, "/_search_arrayformat/template", this);
029        controller.registerHandler(GET, "/{index}/_search_arrayformat/template", this);
030        controller.registerHandler(POST, "/{index}/_search_arrayformat/template", this);
031        controller.registerHandler(GET, "/{index}/{type}/_search_arrayformat/template", this);
032        controller.registerHandler(POST, "/{index}/{type}/_search_arrayformat/template", this);
033    }
034
035    @Override
036    public void handleRequest(final RestRequest request, final RestChannel channel, Client client) {
037        SearchRequest searchRequest;
038        searchRequest = RestSearchAction.parseSearchRequest(request);
039        searchRequest.listenerThreaded(false);
040        client.search(searchRequest, new ArrayformatToXContentListener(channel));
041    }
042
043}
